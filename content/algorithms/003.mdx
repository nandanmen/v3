Given a string, return the length of the longest substring without repeating characters.

```js
function lengthOfLongestSubstring(s) {
  const seen = new Set()
  let max = 0
  let j = 0
  let i = 0

  while (i < s.length && j < s.length) {
    if (!seen.has(s[j])) {
      seen.add(s[j++])
      max = Math.max(max, j - i)
    } else {
      seen.delete(s[i++])
    }
  }

  return max
}
```
